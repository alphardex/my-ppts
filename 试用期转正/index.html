工作内容

						
							import Vue from "vue";

							const me = new Vue({
								data() {
									return {
										name: '王浩彬',
										dept: '技术部',
										post: 'web前端(实习)',
										since: '2020年5月18日',
										atWork: true,
										materials: {} as Materials,
										pages: {} as Pages,
										problems: [] as Problem[]
									}
								},
								methods: {
									async writePages(materials: Materials, problems: Problem[]) {
										const { docs, design, api } = materials;
										await this.read(docs);
										await this.process(design);  // Photoshop, Pxcook, etc
										await this.layout(design);  // HTML, CSS[SCSS]
										await this.call(api);  // JavaScript[Typescript], axios
										for (const problem of problems) {
											await this.solve(problem);  // Layout, Logic, Change, etc
										}
									},
								},
								async mounted() {
									while (this.atWork) {
										await this.getMaterials();
										await this.writePages(this.materials, this.problems);
										await this.upload(this.pages);
									}
								}
							})
